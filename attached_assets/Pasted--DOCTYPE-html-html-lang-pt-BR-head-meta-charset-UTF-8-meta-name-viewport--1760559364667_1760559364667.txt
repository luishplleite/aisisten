    <script>
        // Lógica para manipular o formulário e fazer a chamada de API
        const form = document.getElementById('settingsForm');
        const statusMessage = document.getElementById('statusMessage');
        const submitButton = document.getElementById('submitButton');

        form.addEventListener('submit', async (event) => {
            event.preventDefault(); // Previne o recarregamento da página
            
            // Mostra feedback visual no botão
            submitButton.disabled = true;
            submitButton.textContent = 'Salvando...';
            statusMessage.textContent = '';
            statusMessage.className = 'mt-4 text-center text-sm';

            // Coleta os dados do formulário
            const formData = new FormData(form);
            const apiKey = formData.get('apiKey');
            const instanceName = formData.get('instanceName');
            const apiBaseUrl = formData.get('apiBaseUrlInput'); 

            // Constrói a URL final, garantindo que não haja barras duplas
            const finalApiUrl = (apiBaseUrl.endsWith('/') ? apiBaseUrl : apiBaseUrl + '/') + instanceName;

            // Constrói o corpo (payload) da requisição com os nomes corretos
            const payload = {
                rejectCall: formData.has('rejectCall'),
                msgCall: formData.get('msgCall') || "Chamada rejeitada automaticamente.",
                groupsIgnore: formData.has('groupsIgnore'),
                alwaysOnline: formData.has('alwaysOnline'),
                readMessages: formData.has('readMessages'),
                syncFullHistory: formData.has('syncFullHistory'),
                readStatus: formData.has('readStatus')
            };

            try {
                // Usa a URL construída
                const response = await fetch(finalApiUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'apikey': apiKey
                    },
                    body: JSON.stringify(payload)
                });

                if (response.ok) {
                    // Sucesso
                    statusMessage.textContent = 'Configurações salvas com sucesso!';
                    statusMessage.classList.add('text-green-600');
                } else {
                    // Erro na resposta da API
                    const errorData = await response.json().catch(() => ({ message: 'Verifique o console para detalhes.' }));
                    statusMessage.textContent = `Erro: ${response.status} - ${errorData.message || response.statusText}`;
                    statusMessage.classList.add('text-red-600');
                }
            } catch (error) {
                // Erro de rede ou outro problema
                console.error('Falha na requisição:', error);
                statusMessage.textContent = 'Falha na comunicação com o servidor. Verifique a URL e sua conexão.';
                statusMessage.classList.add('text-red-600');
            } finally {
                // Reabilita o botão
                submitButton.disabled = false;
                submitButton.textContent = 'Salvar Configurações';
            }
        });
    </script>
